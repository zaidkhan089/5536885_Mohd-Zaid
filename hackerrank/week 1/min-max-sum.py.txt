import math
import os
import random
import re
import sys



def miniMaxSum(arr):
    sum1 = float('inf')
    sum2 = float('-inf')
    temp = 0

    for i in range(len(arr)):
        temp = 0
        for j in range(len(arr)):
            if i == j:
                continue
            else:
                temp += arr[j]
        
        
        sum1 = min(sum1, temp)
        sum2 = max(sum2, temp)

    print(sum1, sum2)


        

if __name__ == '__main__':

    arr = list(map(int, input().rstrip().split()))

    miniMaxSum(arr)
