#include <stdio.h>
#include <ctype.h>
#include <string.h>

#define ALPHABET_COUNT 26

char* pangrams(char s[]) {
    int letters[ALPHABET_COUNT] = {0}; 
    int uniqueCount = 0;

    for (int i = 0; s[i] != '\0'; i++) {
        if (isalpha(s[i])) {
            char lower = tolower(s[i]);           
            int index = lower - 'a';              
            if (letters[index] == 0) {          
                letters[index] = 1;
                uniqueCount++;
            }
        }
    }

    if (uniqueCount == ALPHABET_COUNT) {
        return "pangram";
    } else {
        return "not pangram";
    }
}

int main() {
    char s[1000]; // assuming max length of input is 999 chars
    fgets(s, sizeof(s), stdin);

    printf("%s\n", pangrams(s));

    return 0;
}
